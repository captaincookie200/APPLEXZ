local UserIS = game:GetService("UserInputService")

local HotKey = "F"
local Toggle = false

UserIS.InputBegan:Connect(function(LocalInput)
if LocalInput.KeyCode == Enum.KeyCode[HotKey] then
Toggle = not Toggle
while wait() do
if Toggle == true then
mouse1click(MOUSE_CLICK)
else
mouse1click(MOUSE_UP)
end
end
end
end)
